1. 下面这段 C++ 代码，编译时会（ ）。
#include <iostream>
using namespace std;
int main() {
    int a = 10, b = 3;
    int c = a % b;     // 第1行
    int d = a /= b;    // 第2行
    int e = a >= b;    // 第3行
    int f = a => b;    // 第4行
    cout << c << d << e << f;
    return 0;
}
A. 编译通过，输出 1331
B. 第2行出错，/= 不能用于初始化
C. 第4行出错，=> 不是合法运算符
D. 第1行出错，% 不能用于 int 类型

4. 下面 C++ 代码执行后的输出是（ ）。
#include <iostream>
using namespace std;
int main() {
    int n = 4567;
    int ones = n % 10;        // 个位
    int tens = n / 10 % 10;   // 十位
    if ((ones + tens) % 2 == 0)
        cout << "even";
    else
        cout << "odd";
    return 0;
}
A. even，因为 ones=7, tens=6, 和为13是奇数 —— 等等，应该输出 odd
B. even，因为 4567 是奇数
C. even，因为 ones=7, tens=5
D. odd，因为 ones=7, tens=6, 和为13，13%2!=0

10. 下面 C++ 代码执行后的输出是（ ）。
#include <iostream>
using namespace std;
int main() {
    int a = 13, b = 7;
    // 13 = 1101(二进制), 7 = 0111(二进制)
    a ^= b;   // 第1步
    b ^= a;   // 第2步
    a ^= b;   // 第3步
    cout << a << " " << b;
    return 0;
}

B. 10 13（异或后a=10）
C. 0 0（互相抵消）

12. 下面 C++ 代码执行后的输出是（ ）。
#include <iostream>
using namespace std;
int main() {
    int total = 0;
    for (int num = 38; num <= 40; num++) {
        int temp = num, digitSum = 0;
        while (temp > 0) {
            digitSum += temp % 10;
            temp /= 10;
        }
        total += digitSum;
    }
    cout << total;
    return 0;
}
A. 27（digitSum: 11+12+4=27）
B. 30（digitSum: 11+12+7=30）
C. 15（3+8+3+9+4+0=27被看成15）
D. 23（digitSum: 11+8+4=23）
13. 下面 C++ 代码执行后的输出是（ ）。
#include <iostream>
using namespace std;
int main() {
    int count = 0;
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            for (int k = 0; k < 3; k++) {
                if (k == 1) break;
                count++;
            }
        }
    }
    cout << count;
    return 0;
}
A. 3（break终止了所有循环）
B. 9（break只终止最内层，每次内层只执行k=0）
C. 27（break没起作用）
D. 1（第一次break就全部停止）
14. 下面 C++ 代码执行后的输出是（ ）。
#include <iostream>
using namespace std;
int main() {
    int n = 5;
    int count = 0;

        count++;
    }
    cout << count << " " << n;
    return 0;

A. 5 0（循环5次，n最终为0）
B. 4 0（n--先减再判断）
C. 5 -2
D. 5 -1（循环5次，n从5减到-1时停止）
