五、编程题参考程序
编程题 1：求最大值与最小值
#include <iostream>
using namespace std;
int main() {
    int n;
    cin >> n;
    int x;
    cin >> x;
    int mx = x, mn = x;        // 用第一个数初始化
    for (int i = 1; i < n; i++) {
        cin >> x;
        if (x > mx) mx = x;
        if (x < mn) mn = x;
    }
    cout << mx << " " << mn << endl;
    return 0;
}
要点：用读入的第一个数初始化最大/最小值（不要用 0 初始化，否则负数会出错）。
编程题 2：判断回文数
#include <iostream>
using namespace std;
int main() {
    long long N, t, r = 0;
    cin >> N;
    t = N;
    while (t > 0) {            // 把 N 倒过来得到 r
        r = r * 10 + t % 10;
        t /= 10;
    }
    cout << (r == N ? "YES" : "NO") << endl;
    return 0;
}
要点：把 N 逆序得到 r，若 r 与原数相等即回文。注意逆序前先用临时变量 t 保存 N，N 最大 20 亿用 long long。
