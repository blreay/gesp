五、编程题参考程序
编程题 1：时间换算
#include <iostream>
using namespace std;
int main() {
    long long s;
    cin >> s;
    long long h = s / 3600;
    long long m = s % 3600 / 60;
    long long sec = s % 60;
    cout << h << " " << m << " " << sec << endl;
    return 0;
}
要点：小时 = 总秒 /3600；分钟 = 余下秒 /60（先 %3600 再 /60）；秒 = %60。
编程题 2：最大公约数
#include <iostream>
using namespace std;
int main() {
    int a, b;
    cin >> a >> b;
    // 辗转相除法（欧几里得算法）
    while (b != 0) {
        int r = a % b;
        a = b;
        b = r;
    }
    cout << a << endl;
    return 0;
}
要点：辗转相除法——不断用 a % b 取代，直到余数为 0，此时 a 就是最大公约数。也可从 min(a,b) 往下枚举第一个公约数（数据范围小，亦可通过）。
