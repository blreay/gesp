编程题 1：购票找零
#include <iostream>
using namespace std;
int main() {
    long long m;
    cin >> m;
    cout << m / 45 << " " << m % 45 << endl;
    return 0;
}
要点：能买票数 = m / 45（整除），剩余 = m % 45（取模）。
编程题 2：统计奇偶
#include <iostream>
using namespace std;
int main() {
    int n;
    cin >> n;
    int odd = 0, even = 0, oddSum = 0, x;
    for (int i = 0; i < n; i++) {
        cin >> x;
        if (x % 2 != 0) { odd++; oddSum += x; }  // 注意负奇数
        else even++;
    }
    cout << odd << " " << even << " " << oddSum << endl;
    return 0;
}
要点：用 x % 2 != 0 判断奇数（对负数也成立，C++ 中负奇数 %2 得 -1）；偶数（含 0）走 else 分支。
