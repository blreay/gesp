三、编程题参考答案
编程题 1：小杨买书
思路：先各买 1 本，花费 a+b，剩余 n-a-b 元。然后用剩余的钱尽可能多买便宜的那种书。总数 = 2 + 剩余钱 / min(a,b)。如果 a+b > n 则输出 0。 
#include <cstdio>
using namespace std;

int main() {
    int a, b, n;
    scanf("%d %d %d", &a, &b, &n);
    if (a + b > n) {
        printf("0\n");
        return 0;
    }
    int remain = n - a - b;
    int cheaper = a < b ? a : b;
    int total = 2 + remain / cheaper;
    printf("%d\n", total);
    return 0;
}
编程题 2：数字三角形
思路：双重循环，外层 i 从 1 到 n 控制行，内层 j 从 1 到 i 输出数字 i。注意数字之间用空格分隔，行末无多余空格。 
#include <cstdio>
using namespace std;

int main() {
    int n;
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= i; j++) {
            if (j > 1) printf(" ");
            printf("%d", i);
        }
        printf("\n");
    }
    return 0;
}
