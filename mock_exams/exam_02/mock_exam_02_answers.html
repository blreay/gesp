三、编程题参考答案
编程题 1：回文数判断
思路：对 L 到 R 之间的每个数，求其逆序数，如果逆序数等于原数，则为回文数。 
#include <iostream>
using namespace std;

int main() {
    int L, R;
    cin >> L >> R;
    int count = 0;
    for (int n = L; n <= R; n++) {
        int temp = n, rev = 0;
        while (temp > 0) {
            rev = rev * 10 + temp % 10;
            temp /= 10;
        }
        if (rev == n)
            count++;
    }
    cout << count << endl;
    return 0;
}
编程题 2：水仙花数
思路：遍历 A 到 B 之间的每个三位数，分解出百位 a、十位 b、个位 c，
判断 a³+b³+c³ 是否等于该数。 
#include <iostream>
using namespace std;

int main() {
    int A, B;
    cin >> A >> B;
    int found = 0;
    for (int i = A; i <= B; i++) {
        int a = i / 100;         // 百位
        int b = (i / 10) % 10;   // 十位
        int c = i % 10;          // 个位
        if (a*a*a + b*b*b + c*c*c == i) {
            cout << i << endl;
            found = 1;
        }
    }
    if (!found)
        cout << "None" << endl;
    return 0;
}
