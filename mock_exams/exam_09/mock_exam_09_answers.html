三、编程题参考答案
编程题 1：鸡兔同笼
思路：设鸡有 c 只、兔有 r 只，则 c + r = h，2c + 4r = f。解方程得 r = (f - 2h) / 2，c = h - r。需要检查 (f - 2h) 是否为非负偶数，且 c 和 r 均为非负整数。 
#include <iostream>
using namespace std;

int main() {
    int h, f;
    cin >> h >> f;
    int diff = f - 2 * h;
    if (diff < 0 || diff % 2 != 0) {
        cout << "No Solution" << endl;
        return 0;
    }
    int rabbit = diff / 2;
    int chicken = h - rabbit;
    if (chicken < 0) {
        cout << "No Solution" << endl;
        return 0;
    }
    cout << chicken << " " << rabbit << endl;
    return 0;
}
编程题 2：统计字符
思路：读入 n 个整数，逐个判断正数、负数、零并分别计数。 
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int pos = 0, neg = 0, zer = 0;
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        if (x > 0) pos++;
        else if (x < 0) neg++;
        else zer++;
    }
    cout << pos << " " << neg << " " << zer << endl;
    return 0;
}
